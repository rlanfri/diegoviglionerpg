<script>

// Simulacion basica de las funciones requeridas por la UI para evitar congelamientos

let selectedRole = 'paciente';

function selRole(el, role) {

  document.querySelectorAll('.role-pill').forEach(p => p.classList.remove('on'));

  el.classList.add('on');

  selectedRole = role;

}

function toast(msg) {

  alert(msg);

}


// ── CONFIGURACIÓN DE LOGIN POR REDIRECCIÓN ──

async function doLogin() {

  const btn = document.querySelector('.btn-login');

  if (btn) {

    btn.innerHTML = 'Redirigiendo a Google...';

    btn.disabled = true;

  }

  try {

    const { auth, provider, signInWithRedirect } = window._fb;

    await signInWithRedirect(auth, provider);

  } catch (error) {

    console.error("Error al redirigir:", error);

    toast("Error al conectar con Google");

    if (btn) { btn.innerHTML = 'Ingresar con Google'; btn.disabled = false; }

  }

}


// ── DETECTOR INTELIGENTE DE RETORNO CON RESCATE POR EMAIL ──

window.addEventListener('load', () => {

  setTimeout(async () => {

    if (!window._fb || !window._fb.auth) return;


    try {

      const { auth, getRedirectResult, db, doc, getDoc, collection, query, where, getDocs, setDoc, deleteDoc, signOut } = window._fb;

      const result = await getRedirectResult(auth);

      

      if (result && result.user) {

        const firebaseUser = result.user;

        const emailLogin = firebaseUser.email.toLowerCase().trim();


        const btn = document.querySelector('.btn-login');

        if (btn) { btn.innerHTML = 'Verificando...'; btn.disabled = true; }


        // PASO 1: Buscar directo por el UID real de Google (Logins sucesivos)

        const userDocRef = doc(db, 'usuarios', firebaseUser.uid);

        let snap = await getDoc(userDocRef);

        let perfil = null;


        if (snap.exists()) {

          perfil = snap.data();

        } else {

          // PASO 2: RESCATE por Email (Primer login de cuentas creadas con ID aleatorio)

          const q = query(collection(db, 'usuarios'), where('email', '==', emailLogin));

          const querySnap = await getDocs(q);


          if (!querySnap.empty) {

            const docViejo = querySnap.docs[0];

            perfil = docViejo.data();

            const idViejo = docViejo.id;


            // Clonamos los datos usando el UID real de Google como ID de documento definitivo

            await setDoc(doc(db, 'usuarios', firebaseUser.uid), {

              ...perfil,

              uid: firebaseUser.uid

            });


            // Borramos el documento viejo con ID aleatorio de la base de datos

            if (idViejo !== firebaseUser.uid) {

              await deleteDoc(doc(db, 'usuarios', idViejo));

            }

          }

        }


        // PASO 3: Validar estado activo

        if (!perfil || perfil.estado !== 'activo') {

          await signOut(auth);

          window._showAccessError(!perfil ? 'no-registrado' : 'sin-acceso');

          return;

        }


        // PASO 4: Inicializar la interfaz real de la aplicación

        window._initAppWithUser(firebaseUser.uid, perfil.rol, perfil.nombre || firebaseUser.displayName || '');

      }

    } catch (error) {

      console.error("Error al procesar el retorno de autenticación:", error);

      const btn = document.querySelector('.btn-login');

      if (btn) { btn.innerHTML = '<div class="g-dot">G</div>Ingresar con Google'; btn.disabled = false; }

    }

  }, 600);

});


window._showAccessError = function(tipo) {

  const msgs = {

    'no-registrado': 'Tu cuenta no está registrada en el sistema. Contactá al consultorio.',

    'sin-acceso': 'Tu acceso está suspendido. Contactá al consultorio.',

  };

  const card = document.querySelector('.login-card');

  if (card) {

    // Evitamos duplicar carteles de error

    const viejoErr = card.querySelector('.err-msg');

    if (viejoErr) viejoErr.remove();


    const err = document.createElement('div');

    err.className = 'err-msg';

    err.style.cssText = 'background:#fee2e2;color:#991b1b;border-radius:8px;padding:.75rem 1rem;font-size:.82rem;margin-top:.75rem;line-height:1.5;text-align:center;';

    err.textContent = msgs[tipo] || 'Error de acceso.';

    card.appendChild(err);

  }

  const btn = document.querySelector('.btn-login');

  if (btn) { btn.innerHTML = 'Ingresar con Google'; btn.disabled = false; }

};


window._initAppWithUser = function(uid, rol, nombre) {

  document.querySelector('.login-screen').classList.remove('active');

  document.body.classList.remove('login-view');

  document.getElementById('app').classList.add('active');

  // Aquí corre la lógica nativa del dashboard que tenías implementada

  console.log("App cargada para:", nombre, "con rol:", rol);

};
